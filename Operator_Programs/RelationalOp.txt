//programm Using Relation operator
public class RelationaOperator {

	public static void main(String[] args) {
		int num1=43,num2=56;
		System.out.println(num1<num2);
		System.out.println(num1>num2);
		System.out.println(num1<=num2);
		System.out.println(num1>=num2);
		System.out.println(num1==num2);
		System.out.println(num1!=num2);
		
	}

}
